-- V. CONSTRUCTION OF QUERIES

-- C00. For this file we included the comments at the end of the script, as we are refering to many queries at each comment.

-- A. ALL THE RECORDS OF ONE TABLE
-- ---------------------------------------
SELECT * FROM TEAM;

-- B. A PROJECTION WITH 3 COLUMNS IN REVERSE ALPHABETICAL ORDER
-- ---------------------------------------
SELECT ID, NAME, CITY_FOUNDED, GOALS_SCORED
FROM TEAM
ORDER BY CITY_FOUNDED DESC, NAME DESC;

-- C. RESTRICTION QUERIES WITH MULTIPLE CLAUSES
-- ---------------------------------------
-- 1. PLAYERS FROM A SPECIFIC COUNTRY
SELECT FIRST_NAME, SURNAME, COUNTRY 
FROM FOOTBALLER
WHERE COUNTRY='GREECE';

-- 2. BEGINS WITH J, ENDS WITH SON
SELECT FIRST_NAME, SURNAME 
FROM FOOTBALLER
WHERE SURNAME LIKE 'J%SON';

-- 3. ARE NOT FROM ENGLAND
SELECT FIRST_NAME, SURNAME, COUNTRY 
FROM FOOTBALLER
WHERE COUNTRY<>'ENGLAND';

-- 4. BORN FROM APRIL TO SEPTEMBER
SELECT FIRST_NAME, SURNAME, DOB, TO_CHAR(DOB, 'MONTH') 
FROM FOOTBALLER
WHERE TO_NUMBER(TO_CHAR(DOB, 'MM')) BETWEEN 4 AND 9;

-- 5. BRAZIL/ENGLAND, GK/MID, SURNAME STARTS WITH L BUT NOT CONTAINING 'i'
SELECT SURNAME, COUNTRY, POSITION
FROM FOOTBALLER
WHERE COUNTRY IN ('BRAZIL', 'ENGLAND') AND POSITION IN ('GK', 'MID') AND SURNAME LIKE 'L%' AND SURNAME NOT LIKE '%i%'
ORDER BY COUNTRY, POSITION;

-- D. JOIN QUERIES
-- ---------------------------------------
-- 1. TWO TABLES
SELECT T.NAME, T.GOALS_SCORED, F.SURNAME, F.GOALS 
FROM TEAM T, FOOTBALLER F
WHERE T.ID=F.TEAM_ID AND F.COUNTRY IN ('ENGLAND','BRAZIL');

-- 2. THREE TABLES
SELECT S.NAME, S.CITY, T.NAME, T.CITY_FOUNDED, F.SURNAME, F.COUNTRY 
FROM FOOTBALLER F, TEAM T, STADIUM S
WHERE T.ID=F.TEAM_ID AND S.ID=T.STADIUM_ID AND F.CONTRACT>70000 AND COUNTRY NOT IN ('BRAZIL', 'GREECE', 'ENGLAND');

SELECT TM.NAME, SP.NAME, SS.FEE
FROM SPONSORSHIP SS
JOIN TEAM TM ON SS.TEAM_ID = TM.ID
JOIN SPONSOR SP ON SS.SPONSOR_ID = SP.ID
ORDER BY SS.FEE DESC;

-- 3. MORE TABLES
SELECT SP.NAME, TM.NAME, ST.NAME, COUNT(FB.ID), SS.FEE
FROM STADIUM ST, TEAM TM, FOOTBALLER FB, SPONSOR SP, SPONSORSHIP SS
WHERE SP.ID=SS.SPONSOR_ID AND TM.ID=SS.TEAM_ID AND TM.ID=FB.TEAM_ID AND ST.ID=TM.STADIUM_ID 
GROUP BY SP.NAME, TM.NAME, ST.NAME, SS.FEE;

-- E. AGGREGATE FUNCTIONS
-- ---------------------------------------
-- 1. FOOTBALLERS WITH SURNAME 'K...S'
SELECT FIRST_NAME, SURNAME, COUNTRY 
FROM FOOTBALLER
WHERE SURNAME LIKE 'K%S';

-- 2. TOTAL NUMBER OF PLAYERS PER TEAM
SELECT T.NAME, COUNT(*) PLAYERS 
FROM FOOTBALLER F, TEAM T
WHERE T.ID=F.TEAM_ID
GROUP BY T.NAME;

SELECT T.ID, T.NAME, COUNT(F.SURNAME) NUMBER_OF_PLAYERS
FROM FOOTBALLER F
JOIN TEAM T ON F.TEAM_ID = T.ID
GROUP BY T.NAME, T.ID;

-- 3. HIGHEST PAID FOOTBALLER
SELECT SURNAME, FIRST_NAME, CONTRACT
FROM FOOTBALLER
WHERE CONTRACT=(SELECT MAX(CONTRACT) FROM FOOTBALLER);

-- 4. HOW MANY STADIUMS WITH REVIEW SCORE LESS THAN 2
SELECT COUNT(*) COUNT_RS_LESS_2 
FROM STADIUM
WHERE REVIEW_SCORE<2;

-- 5. SURNAME, FIRST_NAME, ID AND THEIR TEAM IN ASCENDING ORDER
SELECT F.SURNAME, F.FIRST_NAME, F.ID, T.NAME 
FROM FOOTBALLER F, TEAM T
WHERE T.ID=F.TEAM_ID
ORDER BY SURNAME ASC, FIRST_NAME ASC;

-- 6. FOOTBALLER (SURNAME, NAME, TEAM) HAS THE MOST GOALS
SELECT F.SURNAME, F.FIRST_NAME, T.NAME, F.GOALS 
FROM FOOTBALLER F, TEAM T
WHERE T.ID=F.TEAM_ID AND F.GOALS=(SELECT MAX(GOALS) FROM FOOTBALLER);

-- 7. TOTAL NUMBER OF TEAMS PER CITY
SELECT CITY_FOUNDED, COUNT(*) TEAMS_PER_CITY 
FROM TEAM
GROUP BY CITY_FOUNDED
ORDER BY COUNT(*) DESC, CITY_FOUNDED ASC;

-- 8. TEAM WITH THE WORST GOAL DIFFERENCE
SELECT NAME, GOALS_SCORED-GOALS_CONCEDED DIFF 
FROM TEAM
WHERE (GOALS_SCORED-GOALS_CONCEDED)=(SELECT MIN(GOALS_SCORED-GOALS_CONCEDED) FROM TEAM);

-- 9. NUMBER OF STADIUMS BUILT BETWEEN 1980 AND 1999
SELECT COUNT(*) COUNT_YEARBUILT_1980_1999 
FROM STADIUM
WHERE YEAR_BUILT BETWEEN 1980 AND 1999;

SELECT COUNT(NAME) AS COUNT_OF_STADIUMS_1980_1999
FROM STADIUM
WHERE YEAR_BUILT >= 1980 AND YEAR_BUILT <= 1999;

-- 10. PLAYERS THAT WERE BORN IN JULY
SELECT FIRST_NAME, SURNAME, COUNTRY, DOB 
FROM FOOTBALLER
WHERE TO_CHAR(DOB, 'MON')='JUL'
ORDER BY TO_CHAR(DOB,'DD') ASC;

SELECT FIRST_NAME, SURNAME, DOB
FROM FOOTBALLER
WHERE TO_NUMBER(TO_CHAR(DOB, 'MM')) = 7;

-- 11. NUMBER OF STADIUMS THAT DON'T HOST A TEAM
SELECT COUNT(*) NUMBER_OF_STADIUMS_NOT_HOSTING_TEAMS
FROM STADIUM S
WHERE S.ID NOT IN (SELECT DISTINCT T.STADIUM_ID FROM TEAM T);

-- 12. TEAM WITH NAME (%SAINT%)
SELECT NAME 
FROM TEAM
WHERE NAME LIKE '%SAINT%';

-- 13. PLAYER FROM BRAZIL WHO HAS THE HIGHEST CONTRACT OR SCORED MOST GOALS
SELECT FIRST_NAME, SURNAME, CONTRACT, GOALS 
FROM FOOTBALLER
WHERE COUNTRY='BRAZIL' AND 
    (CONTRACT=(SELECT MAX(CONTRACT) FROM FOOTBALLER WHERE COUNTRY='BRAZIL') OR GOALS=(SELECT MAX(GOALS) FROM FOOTBALLER WHERE COUNTRY='BRAZIL'));
-- VERIFICATION
SELECT FIRST_NAME, SURNAME, CONTRACT, GOALS 
FROM FOOTBALLER
WHERE COUNTRY='BRAZIL';

-- 14. PLAYERS WITH HIGHEST CONTRACT PER POSITION
SELECT F.SURNAME, F.FIRST_NAME, F.POSITION, F.CONTRACT
FROM FOOTBALLER F, (SELECT POSITION POS, MAX(CONTRACT) MAXCON FROM FOOTBALLER GROUP BY POSITION) M
WHERE F.POSITION=POS AND F.CONTRACT=MAXCON;

SELECT F.FIRST_NAME, F.SURNAME, F.POSITION, F.CONTRACT
FROM FOOTBALLER F
WHERE F.CONTRACT = (
    SELECT MAX(F2.CONTRACT)
    FROM FOOTBALLER F2
    WHERE F2.POSITION = F.POSITION)
ORDER BY F.POSITION;

-- 15. USERNAME = <2 FIRST COUNTRY> + <4 FIRST FIRST_NAME> + <4 FIRST SURNAME>
SELECT ID, SUBSTR(COUNTRY,1,2)||SUBSTR(FIRST_NAME,1,4)||SUBSTR(SURNAME,1,4) USERNAME
FROM FOOTBALLER;

-- 16. TOTAL OF FOOTBALLERS' CONTRACTS OF THE LEAGUE
SELECT SUM(CONTRACT) TOTAL_CONTRACT_AMOUNT FROM FOOTBALLER;

-- 17. SPONSOR SPENDING MOST MONEY
SELECT SP.NAME, SUM(SS.FEE) 
FROM SPONSOR SP, SPONSORSHIP SS
WHERE SP.ID=SS.SPONSOR_ID
GROUP BY SP.NAME
HAVING SUM(SS.FEE)=(SELECT MAX(SUMFEE) FROM (SELECT SUM(SS.FEE) SUMFEE FROM SPONSORSHIP SS GROUP BY SS.SPONSOR_ID) M);

-- 18. TYPE OF BUSINESS THAT SPENDS MOST MONEY
SELECT SP.TYPE, SUM(SS.FEE)
FROM SPONSOR SP, SPONSORSHIP SS
WHERE SP.ID=SS.SPONSOR_ID
GROUP BY SP.TYPE
HAVING SUM(SS.FEE)=(SELECT MAX(XSUMFEE) FROM (
    SELECT SUM(SS.FEE) XSUMFEE FROM SPONSOR SP, SPONSORSHIP SS WHERE SP.ID=SS.SPONSOR_ID GROUP BY SP.TYPE) X);

-- 19. FOOTBALLERS WITH HIGHEST AVERAGE CONTRACTS PER SPONSOR
SELECT FB.FIRST_NAME, FB.SURNAME, TM.NAME, FB.CONTRACT, SP.NAME
FROM FOOTBALLER FB, TEAM TM, SPONSOR SP, SPONSORSHIP SS
WHERE FB.TEAM_ID=TM.ID AND SP.ID=SS.SPONSOR_ID AND TM.ID=SS.TEAM_ID AND SP.ID IN (
    SELECT SP.ID FROM FOOTBALLER FB, TEAM TM, SPONSOR SP, SPONSORSHIP SS
    WHERE FB.TEAM_ID=TM.ID AND SP.ID=SS.SPONSOR_ID AND TM.ID=SS.TEAM_ID
    GROUP BY SP.ID
    HAVING AVG(FB.CONTRACT)=(SELECT MAX(AVGCON) FROM (
	SELECT AVG(FB.CONTRACT) AVGCON FROM FOOTBALLER FB, TEAM TM, SPONSOR SP, SPONSORSHIP SS
	WHERE FB.TEAM_ID=TM.ID AND SP.ID=SS.SPONSOR_ID AND TM.ID=SS.TEAM_ID
	GROUP BY SP.ID)));
--VERIFICATION
SELECT SP.NAME, AVG(CONTRACT)
FROM FOOTBALLER FB, TEAM TM, SPONSOR SP, SPONSORSHIP SS
WHERE FB.TEAM_ID=TM.ID AND SP.ID=SS.SPONSOR_ID AND TM.ID=SS.TEAM_ID
GROUP BY SP.NAME
ORDER BY AVG(CONTRACT) DESC;

-- 20. TEAM THAT GETS MOST FROM ELECTRONICS SPONSORS
SELECT TM.NAME, SUM(NVL(SS.FEE,0))
FROM TEAM TM, SPONSOR SP, SPONSORSHIP SS
WHERE TM.ID=SS.TEAM_ID AND SP.ID=SS.SPONSOR_ID AND SP.TYPE='ELECTRONICS'
GROUP BY TM.NAME
HAVING SUM(SS.FEE)=(SELECT MAX(XSUMFEE) FROM (SELECT SUM(NVL(SS.FEE,0)) XSUMFEE
    FROM TEAM TM, SPONSOR SP, SPONSORSHIP SS
    WHERE TM.ID=SS.TEAM_ID AND SP.ID=SS.SPONSOR_ID AND SP.TYPE='ELECTRONICS'
    GROUP BY TM.NAME) X);
-- VERIFICATION
SELECT TM.NAME, SUM(NVL(SS.FEE,0))
FROM TEAM TM, SPONSOR SP, SPONSORSHIP SS
WHERE TM.ID=SS.TEAM_ID AND SP.ID=SS.SPONSOR_ID AND SP.TYPE='ELECTRONICS'
GROUP BY TM.NAME
ORDER BY SUM(NVL(SS.FEE,0)) DESC;
