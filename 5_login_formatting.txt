-- A. FORMATTING
-- ---------------------------------------

-- C02. We created two formatting procedures. The second is a wrapper procedure that calls the first one. 
--      So we call only the second. It is important to note that the output of the procedure is not executed code but a text which we can copy, 
--      paste and run it to format our current session.

-- 1. PROCEDURES
CREATE OR REPLACE PROCEDURE PRC_COLS_FORMAT(TAB_NAME VARCHAR2) AS
    QUERY_MAX_COLS_LEN VARCHAR2(200);  
    MAX_COLS_LEN NUMBER;
    FORMATTING VARCHAR2(20);
BEGIN
    DBMS_OUTPUT.PUT_LINE('-- FORMATTING COLUMNS OF ' || TAB_NAME || ':');
    FOR COLS IN (SELECT COLUMN_NAME AS NAME, DATA_TYPE AS TYPE FROM USER_TAB_COLUMNS WHERE TABLE_NAME = UPPER(TAB_NAME) ORDER BY COLUMN_ID) LOOP
        QUERY_MAX_COLS_LEN := 'SELECT MAX(LENGTH(' || COLS.NAME || ')) FROM ' || TAB_NAME;
        EXECUTE IMMEDIATE QUERY_MAX_COLS_LEN INTO MAX_COLS_LEN;
	FORMATTING:=CASE 
	    WHEN COLS.TYPE='NUMBER' THEN LPAD('9', MAX_COLS_LEN, '9') 
	    ELSE 'A'||TO_CHAR(MAX_COLS_LEN)
	END;
        DBMS_OUTPUT.PUT_LINE('COLUMN '||COLS.NAME||' FORMAT '||FORMATTING);
    END LOOP;
END;
/

CREATE OR REPLACE PROCEDURE PRC_START AS
BEGIN
    DBMS_OUTPUT.PUT_LINE('
ALTER SESSION SET NLS_DATE_FORMAT = ''DD-MON-YYYY'';

-- FORMATTING SESSION
SET LINESIZE 200
SET PAGESIZE 50000
SET WRAP OFF
SET WORD_WRAPPED OFF
SET NEWPAGE 0
SET SPACE 1
');
    FOR TABS IN (SELECT TABLE_NAME AS NAME FROM USER_TABLES ORDER BY TABLE_NAME) LOOP
	PRC_COLS_FORMAT(TABS.NAME);
	DBMS_OUTPUT.PUT_LINE('
');
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('CLEAR SCREEN');
END;
/
